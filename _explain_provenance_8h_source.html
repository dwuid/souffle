54  indices.push_back(idx);
55  }

57  /** Verify if elements at the indices are equivalent in the given product */
58  bool verify(const std::vector<tuple>& product) const {
59  for (size_t i = 1; i < indices.size(); ++i) {
60  if (product[indices[i].first][indices[i].second] !=
61  product[indices[i - 1].first][indices[i - 1].second]) {
62  return false;
63  }
64  }
65  return true;
66  }
67

69  std::pair<size_t, size_t> getFirstIdx() {
70  return indices[0];
71  }
72

74  std::vector<std::pair<size_t, size_t>> getIndices() {
75  return indices;
76  }
77 
78  /** Get type of the variable of the equivalence class, 'i' for RamDomain, 's' for symbol */
79  char getType() {

81  }
82

127 /** utility function to split a string */
128 inline std::vector<std::string> split(const std::string& s, char delim, int times = -1) {
129  std::vector<std::string> v;
130  std::stringstream ss(s);
131  std::string item;
132 
133  while ((times > 0 || times <= -1) && std::getline(ss, item, delim)) {
134  v.push_back(item);
135  times--;
136  }
137 
138  if (ss.peek() != EOF) {
139  std::string remainder;
140  std::getline(ss, remainder);
141  v.push_back(remainder);
142  }
143 
144  return v;
145 }
